```
SELECT
RECIPE.NAME,
RECIPE_INGREDIENT.AMOUNT AS REQUIRED_AMOUNT,
INGREDIENT.NAME,
INGREDIENT.MEASURE,
INVENTORY_ITEM.AMOUNT AS INVENTORY_AMOUNT,
INVENTORY_ITEM.EXPIRATION
FROM RECIPE
LEFT JOIN RECIPE_INGREDIENT ON RECIPE_INGREDIENT.RECIPE_UUID=RECIPE.UUID
LEFT JOIN INGREDIENT ON RECIPE_INGREDIENT.INGREDIENT_UUID=INGREDIENT.UUID
LEFT JOIN INVENTORY_ITEM ON INVENTORY_ITEM.INGREDIENT_UUID=INGREDIENT.UUID;
```
